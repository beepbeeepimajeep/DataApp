\[
v_2(30!)=\left\lfloor \frac{30}{2}\right\rfloor+\left\lfloor \frac{30}{4}\right\rfloor+\left\lfloor \frac{30}{8}\right\rfloor+\left\lfloor \frac{30}{16}\right\rfloor=15+7+3+1=26
\]

\[
v_3(30!)=10+3+1=14,\quad v_5(30!)=6+1=7
\]

\[
v_7=4,\quad v_{11}=2,\quad v_{13}=2,\quad v_{17}=v_{19}=v_{23}=v_{29}=1
\]

\[
30!=2^{26}\cdot 3^{14}\cdot 5^7\cdot 7^4\cdot 11^2\cdot 13^2\cdot 17\cdot 19\cdot 23\cdot 29
